### 1. `backend/routers/credits.py` — 動的価格モデル対応

```python
from fastapi import APIRouter
from models.credits import (
    KPIPayload,
    CreditPriceResponse,
    CreditPurchaseRequest,
    CreditPurchaseResponse,
    CreditSellRequest,
    CreditSellResponse,
    CreditPortfolio,
)

router = APIRouter()


def _base_prices() -> CreditPriceResponse:
    return CreditPriceResponse(
        Biosecurity=14.0,
        Carbon=10.0,
        FoodLoss=9.0,
        Quality=8.0,
        Stability=7.0,
    )


def _adj_factor(demand_index: float, liquidity_index: float, volatility_index: float) -> float:
    # AdjFactor = 1 + 0.3*Demand + 0.2*Liquidity - 0.2*Volatility
    return 1.0 + 0.3 * demand_index + 0.2 * liquidity_index - 0.2 * volatility_index


@router.get("/price", response_model=CreditPriceResponse)
def get_credit_prices(
    demand_index: float = 0.5,
    liquidity_index: float = 0.5,
    volatility_index: float = 0.3,
):
    base = _base_prices()
    adj = _adj_factor(demand_index, liquidity_index, volatility_index)

    return CreditPriceResponse(
        Biosecurity=base.Biosecurity * adj,
        Carbon=base.Carbon * adj,
        FoodLoss=base.FoodLoss * adj,
        Quality=base.Quality * adj,
        Stability=base.Stability * adj,
    )
```

（`buy` / `sell` はそのままでもよいですが、必要なら同じ `adj` を掛ける形に拡張可能です）

### 3. `frontend/src/pages/Dashboard/MarketView.tsx` — 動的パラメータ UI ✔

```tsx
import React, { useEffect, useState } from "react";
import { getCreditMarket } from "../../api/dashboard";
import { getCreditPrices } from "../../api/credits";
import { CreditsMarketTable } from "../../components/tables/CreditsMarketTable";

export default function MarketView() {
  const [market, setMarket] = useState<any | null>(null);
  const [prices, setPrices] = useState<any | null>(null);

  const [params, setParams] = useState({
    demand_index: 0.5,
    liquidity_index: 0.5,
    volatility_index: 0.3,
  });

  useEffect(() => {
    getCreditMarket().then(setMarket);
  }, []);

  useEffect(() => {
    getCreditPrices(params).then(setPrices);
  }, [params]);

  const update = (key: string, value: number) => {
    setParams(prev => ({ ...prev, [key]: value }));
  };

  if (!market || !prices) return null;

  const merged = market.credits.map((c: any) => ({
    ...c,
    price_usd:
      prices[
        c.type.charAt(0).toUpperCase() + c.type.slice(1) as keyof typeof prices
      ],
    value_usd:
      c.volume_pbpe *
      prices[
        c.type.charAt(0).toUpperCase() + c.type.slice(1) as keyof typeof prices
      ],
  }));

  return (
    <div>
      <h2>PBPE Credits Market</h2>

      <div style={{ display: "flex", gap: 16, marginBottom: 16 }}>
        <div>
          <label>Demand Index</label>
          <input
            type="number"
            step="0.1"
            min={0}
            max={1}
            value={params.demand_index}
            onChange={e => update("demand_index", parseFloat(e.target.value))}
          />
        </div>
        <div>
          <label>Liquidity Index</label>
          <input
            type="number"
            step="0.1"
            min={0}
            max={1}
            value={params.liquidity_index}
            onChange={e => update("liquidity_index", parseFloat(e.target.value))}
          />
        </div>
        <div>
          <label>Volatility Index</label>
          <input
            type="number"
            step="0.1"
            min={0}
            max={1}
            value={params.volatility_index}
            onChange={e =>
              update("volatility_index", parseFloat(e.target.value))
            }
          />
        </div>
      </div>

      <CreditsMarketTable data={merged} />
    </div>
  );
}
```
