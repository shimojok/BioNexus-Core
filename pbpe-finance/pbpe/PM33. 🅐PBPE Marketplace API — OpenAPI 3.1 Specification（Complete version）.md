# **PBPE Marketplace API — OpenAPI 3.1**

```yaml
openapi: 3.1.0
info:
  title: PBPE Marketplace API
  version: "1.0.0"
  description: >
    PBPE Marketplace API enables listing, issuing, trading, settling,
    and verifying PBPE-based financial products.  
    It integrates directly with Planetary Health OS (Intent, Source, Webhook, Connect).

servers:
  - url: https://api.pbpe.marketplace/v1

paths:

  /products:
    get:
      summary: List all PBPE financial products
      operationId: listProducts
      responses:
        "200":
          description: List of PBPE products
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: "#/components/schemas/PBPEProduct"

    post:
      summary: Create a new PBPE financial product (issuer only)
      operationId: createProduct
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: "#/components/schemas/PBPEProductInput"
      responses:
        "201":
          description: Product created
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEProduct"

  /products/{product_id}:
    get:
      summary: Get details of a PBPE financial product
      operationId: getProduct
      parameters:
        - in: path
          name: product_id
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Product details
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEProduct"

  /orders:
    post:
      summary: Place a buy/sell order for a PBPE product
      operationId: placeOrder
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: "#/components/schemas/PBPEOrderInput"
      responses:
        "201":
          description: Order placed
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEOrder"

  /orders/{order_id}:
    get:
      summary: Retrieve order status
      operationId: getOrder
      parameters:
        - in: path
          name: order_id
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Order details
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEOrder"

  /market/ticker:
    get:
      summary: Get real-time PBPE market ticker
      operationId: getTicker
      responses:
        "200":
          description: Market ticker
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPETicker"

  /settlements/{product_id}:
    post:
      summary: Trigger settlement for a PBPE product
      operationId: settleProduct
      parameters:
        - in: path
          name: product_id
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Settlement executed
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPESettlement"

  /registry/{pbpe_id}:
    get:
      summary: Retrieve PBPE registry entry
      operationId: getRegistryEntry
      parameters:
        - in: path
          name: pbpe_id
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Registry entry
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPERegistryEntry"

  /portfolio:
    get:
      summary: Get investor PBPE portfolio
      operationId: getPortfolio
      responses:
        "200":
          description: Portfolio details
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEPortfolio"

  /portfolio/impact:
    get:
      summary: Get impact summary of investor portfolio
      operationId: getPortfolioImpact
      responses:
        "200":
          description: Impact summary
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/PBPEImpactSummary"

components:

  schemas:

    PBPEProduct:
      type: object
      properties:
        product_id:
          type: string
        product_type:
          type: string
          enum: [bond, credit, insurance, protection]
        underlying_intent:
          type: string
          enum: [agri, climate, health]
        pbpe_source:
          type: string
        expected_pbpe_yield:
          type: number
        risk_class:
          type: string
          enum: [A, B, C, D]
        duration_months:
          type: integer
        payout_model:
          type: string
          enum: [fixed, variable, hybrid]
        distribution_model:
          type: string
          example: "40/40/20"
        settlement_rules:
          type: object

    PBPEProductInput:
      allOf:
        - $ref: "#/components/schemas/PBPEProduct"

    PBPEOrder:
      type: object
      properties:
        order_id:
          type: string
        product_id:
          type: string
        side:
          type: string
          enum: [buy, sell]
        quantity:
          type: number
        price:
          type: number
        status:
          type: string
          enum: [pending, filled, cancelled]

    PBPEOrderInput:
      type: object
      properties:
        product_id:
          type: string
        side:
          type: string
        quantity:
          type: number
        price:
          type: number

    PBPETicker:
      type: object
      properties:
        product_id:
          type: string
        last_price:
          type: number
        pbpe_yield:
          type: number
        volume_24h:
          type: number

    PBPESettlement:
      type: object
      properties:
        product_id:
          type: string
        pbpe_generated:
          type: number
        distribution:
          type: object
          properties:
            actor:
              type: number
            fund:
              type: number
            platform:
              type: number

    PBPERegistryEntry:
      type: object
      properties:
        pbpe_id:
          type: string
        source_intent:
          type: string
        timestamp:
          type: string
        metadata:
          type: object

    PBPEPortfolio:
      type: object
      properties:
        investor_id:
          type: string
        holdings:
          type: array
          items:
            $ref: "#/components/schemas/PBPEProduct"

    PBPEImpactSummary:
      type: object
      properties:
        total_pbpe:
          type: number
        co2_reduction:
          type: number
        soil_regeneration:
          type: number
        health_improvement:
          type: number
```
