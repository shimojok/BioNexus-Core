```python
"""
Questionnaire Engine for HealthBook
Parses and analyzes 200-item questionnaire data
"""

import json
from typing import Dict, List, Optional
from pathlib import Path


class QuestionnaireEngine:
    """Engine for loading and analyzing questionnaire data"""
    
    def __init__(self, data_path: Optional[Path] = None):
        if data_path is None:
            data_path = Path(__file__).parent.parent.parent / "data" / "questionnaire_full.json"
        
        self.data_path = data_path
        self.questions = {}
        self.categories = {}
        self._load_data()
    
    def _load_data(self):
        """Load questionnaire data from JSON file"""
        if self.data_path.exists():
            with open(self.data_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
                self.questions = data.get('questions', {})
                self.categories = data.get('categories', {})
                self.total_questions = data.get('total_questions', 200)
        else:
            print(f"Warning: Data file not found at {self.data_path}")
            self._load_default_questions()
    
    def _load_default_questions(self):
        """Load minimal default questions if file not found"""
        self.questions = {
            "1": {
                "question": "朝食を抜くことがよくある",
                "risk_factors": ["欠食型（朝食抜き）"],
                "related_diseases": ["糖尿病", "肥満"],
                "weight": 0.85
            }
        }
    
    def get_question(self, q_id: int) -> Optional[Dict]:
        """Get a single question by ID"""
        return self.questions.get(str(q_id))
    
    def get_questions_by_category(self, category: str) -> Dict:
        """Get all questions in a category"""
        if category not in self.categories:
            return {}
        
        start, end = self.categories[category]
        return {k: v for k, v in self.questions.items() 
                if start <= int(k) <= end}
    
    def get_all_questions(self) -> Dict:
        """Get all questions"""
        return self.questions
    
    def calculate_risk_factors(self, answers: Dict[int, bool]) -> Dict[str, float]:
        """
        Calculate risk factor scores from answers
        
        Args:
            answers: Dictionary mapping question ID to boolean answer
            
        Returns:
            Dictionary mapping risk factor to accumulated weight
        """
        risk_scores = {}
        
        for q_id, answer in answers.items():
            if not answer:
                continue
            
            q_data = self.questions.get(str(q_id))
            if not q_data:
                continue
            
            weight = q_data.get('weight', 0.5)
            for factor in q_data.get('risk_factors', []):
                risk_scores[factor] = risk_scores.get(factor, 0) + weight
        
        return risk_scores
```

```python
"""
Disease Risk Calculator
Computes disease risk scores based on questionnaire answers
"""

import json
from typing import Dict, List, Optional
from pathlib import Path


class DiseaseRiskCalculator:
    """Calculate disease risk based on questionnaire answers"""
    
    def __init__(self, data_path: Optional[Path] = None):
        if data_path is None:
            data_path = Path(__file__).parent.parent.parent / "data" / "questionnaire_full.json"
        
        self.data_path = data_path
        self.diseases = {}
        self._load_data()
    
    def _load_data(self):
        """Load disease data from JSON file"""
        if self.data_path.exists():
            with open(self.data_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
                self.diseases = data.get('diseases', {})
    
    def calculate_risk(self, risk_factors: Dict[str, float]) -> List[Dict]:
        """
        Calculate disease risk scores
        
        Args:
            risk_factors: Dictionary of risk factors and their accumulated weights
            
        Returns:
            List of disease risk predictions sorted by risk score
        """
        predictions = []
        
        for disease_name, disease_data in self.diseases.items():
            risk_score = 0.0
            matched_factors = []
            max_possible = 0.0
            
            disease_factors = disease_data.get('risk_factors', {})
            
            for factor, weight in disease_factors.items():
                max_possible += weight
                if factor in risk_factors:
                    risk_score += weight
                    matched_factors.append(factor)
            
            # Normalize to 0-100 scale
            if max_possible > 0:
                normalized_score = (risk_score / max_possible) * 100
            else:
                normalized_score = 0.0
            
            if normalized_score > 0:
                predictions.append({
                    'disease': disease_name,
                    'risk_score': round(normalized_score, 1),
                    'risk_level': self._get_risk_level(normalized_score),
                    'matched_factors': matched_factors,
                    'global_prevalence': disease_data.get('global_prevalence', '不明'),
                    'gates_priority': disease_data.get('gates_priority', 'MEDIUM')
                })
        
        # Sort by risk score descending
        predictions.sort(key=lambda x: x['risk_score'], reverse=True)
        return predictions
    
    def _get_risk_level(self, score: float) -> str:
        """Convert risk score to risk level"""
        if score >= 70:
            return "HIGH"
        elif score >= 40:
            return "MEDIUM"
        elif score >= 20:
            return "LOW"
        else:
            return "VERY_LOW"
    
    def get_disease_info(self, disease_name: str) -> Optional[Dict]:
        """Get detailed information about a disease"""
        return self.diseases.get(disease_name)
```

```python
"""
HealthBook Inference Engine
Core engine for metabolic risk assessment and personalized recommendations
"""

from typing import Dict, List, Optional
from pathlib import Path

from .questionnaire_engine import QuestionnaireEngine
from .disease_risk import DiseaseRiskCalculator


class HealthBookInferenceEngine:
    """
    Main inference engine for HealthBook
    Integrates questionnaire analysis and disease risk prediction
    """
    
    def __init__(self, data_path: Optional[Path] = None):
        self.questionnaire_engine = QuestionnaireEngine(data_path)
        self.disease_risk_calculator = DiseaseRiskCalculator(data_path)
    
    def analyze_health(self, answers: Dict[int, bool]) -> Dict:
        """
        Complete health analysis based on questionnaire answers
        
        Args:
            answers: Dictionary mapping question ID (1-200) to boolean answer
            
        Returns:
            Comprehensive health analysis results
        """
        # Step 1: Calculate risk factors from answers
        risk_factors = self.questionnaire_engine.calculate_risk_factors(answers)
        
        # Step 2: Calculate disease risks
        disease_risks = self.disease_risk_calculator.calculate_risk(risk_factors)
        
        # Step 3: Generate summary
        summary = self._generate_summary(disease_risks, risk_factors)
        
        # Step 4: Generate recommendations
        recommendations = self._generate_recommendations(disease_risks)
        
        return {
            'risk_factors': risk_factors,
            'disease_risks': disease_risks[:10],  # Top 10 risks
            'summary': summary,
            'recommendations': recommendations,
            'total_questions_analyzed': len(answers),
            'confidence': 'HIGH' if len(answers) >= 50 else 'MEDIUM'
        }
    
    def _generate_summary(self, disease_risks: List[Dict], risk_factors: Dict) -> Dict:
        """Generate summary of health analysis"""
        high_risks = [d for d in disease_risks if d['risk_level'] == 'HIGH']
        medium_risks = [d for d in disease_risks if d['risk_level'] == 'MEDIUM']
        
        top_risk = disease_risks[0] if disease_risks else None
        
        return {
            'total_diseases_analyzed': len(disease_risks),
            'high_risk_count': len(high_risks),
            'medium_risk_count': len(medium_risks),
            'top_risk_disease': top_risk['disease'] if top_risk else None,
            'top_risk_score': top_risk['risk_score'] if top_risk else 0,
            'total_risk_factors_identified': len(risk_factors)
        }
    
    def _generate_recommendations(self, disease_risks: List[Dict]) -> List[Dict]:
        """Generate personalized recommendations"""
        recommendations = []
        
        for disease in disease_risks[:3]:  # Top 3 risks
            if disease['risk_level'] in ['HIGH', 'MEDIUM']:
                recommendations.append({
                    'disease': disease['disease'],
                    'risk_level': disease['risk_level'],
                    'recommendation': self._get_recommendation_text(disease['disease']),
                    'priority': 'HIGH' if disease['risk_level'] == 'HIGH' else 'MEDIUM'
                })
        
        return recommendations
    
    def _get_recommendation_text(self, disease: str) -> str:
        """Get recommendation text for a disease"""
        recommendations = {
            '糖尿病': '規則正しい食事、適度な運動、糖質制限を心がけてください。',
            '高血圧': '減塩、適度な運動、ストレス管理を心がけてください。',
            '肥満': '食事量の適正化、運動習慣の確立を心がけてください。'
        }
        return recommendations.get(disease, '生活習慣の見直しをお勧めします。')
    
    def get_questionnaire(self) -> Dict:
        """Get all questionnaire questions"""
        return self.questionnaire_engine.get_all_questions()
    
    def get_question(self, q_id: int) -> Optional[Dict]:
        """Get a single question by ID"""
        return self.questionnaire_engine.get_question(q_id)
```
