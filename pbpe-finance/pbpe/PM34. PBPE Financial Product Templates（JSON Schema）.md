🅑 PBPE Financial Product Templates（JSON Schema） を作成してください。

了解、Kaz。  
ここでは **PBPE Financial Product Templates（JSON Schema）** を、  
そのまま `schemas/pbpe-financial-products.json` に置けるレベルでまとめて定義します。

---

### 🧩 全体構成

- **Base:** `PBPEProductBase`（全ての金融商品の共通フィールド）
- **Specialized:**
    - `PBPECarbonIncomeBond`
    - `PBPERegenerativeAgricultureBond`
    - `PBPEHealthImpactBond`
    - `PBPECredit`
    - `PBPEInsurance`
    - `PBPEProtectionProduct`

---

```json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "$id": "https://bionexus.earth/schemas/pbpe-financial-products.json",
  "title": "PBPE Financial Products Schema",
  "description": "Standardized JSON Schemas for PBPE-based financial products (PBPE-FIN-001 family).",
  "type": "object",
  "definitions": {

    "PBPEProductBase": {
      "type": "object",
      "required": [
        "product_id",
        "product_type",
        "underlying_intent",
        "pbpe_source",
        "expected_pbpe_yield",
        "risk_class",
        "duration_months",
        "payout_model",
        "distribution_model",
        "settlement_rules"
      ],
      "properties": {
        "product_id": {
          "type": "string",
          "description": "Unique identifier for the PBPE financial product.",
          "examples": ["pbpe_fin_0001"]
        },
        "product_type": {
          "type": "string",
          "enum": ["bond", "credit", "insurance", "protection"],
          "description": "High-level category of the PBPE financial product."
        },
        "underlying_intent": {
          "type": "string",
          "enum": ["agri", "climate", "health"],
          "description": "Primary Intent type powering this product."
        },
        "pbpe_source": {
          "type": "string",
          "description": "Primary PBPE source metric (e.g., soil, co2, mbt55, medical_cost, yield, water).",
          "examples": ["soil", "co2", "mbt55", "medical_cost", "yield", "water"]
        },
        "expected_pbpe_yield": {
          "type": "number",
          "description": "Expected PBPE yield over the full duration of the product."
        },
        "risk_class": {
          "type": "string",
          "enum": ["A", "B", "C", "D"],
          "description": "PBPE risk rating class."
        },
        "duration_months": {
          "type": "integer",
          "minimum": 1,
          "description": "Product duration in months."
        },
        "payout_model": {
          "type": "string",
          "enum": ["fixed", "variable", "hybrid"],
          "description": "Payout model for the product."
        },
        "distribution_model": {
          "type": "string",
          "description": "Value distribution model, typically '40/40/20'.",
          "examples": ["40/40/20"]
        },
        "currency": {
          "type": "string",
          "description": "Fiat currency used for pricing and settlement.",
          "examples": ["USD", "EUR", "JPY"]
        },
        "issuer": {
          "type": "string",
          "description": "Issuer identifier (organization, cooperative, government, etc.)."
        },
        "region": {
          "type": "string",
          "description": "Geographical region or jurisdiction.",
          "examples": ["JP", "EU", "KE", "BR"]
        },
        "metadata": {
          "type": "object",
          "description": "Additional product metadata.",
          "additionalProperties": true
        },
        "settlement_rules": {
          "type": "object",
          "description": "Rules governing settlement, maturity, and PBPE realization.",
          "properties": {
            "maturity_condition": {
              "type": "string",
              "description": "Condition under which the product is considered matured."
            },
            "pbpe_realization_method": {
              "type": "string",
              "description": "How PBPE is realized (e.g., 'intent_settled', 'threshold_reached')."
            },
            "early_exit_policy": {
              "type": "string",
              "description": "Rules for early redemption or exit."
            }
          },
          "additionalProperties": true
        }
      },
      "additionalProperties": false
    },

    "PBPECarbonIncomeBond": {
      "allOf": [
        { "$ref": "#/definitions/PBPEProductBase" },
        {
          "type": "object",
          "properties": {
            "product_type": {
              "const": "bond"
            },
            "underlying_intent": {
              "const": "climate"
            },
            "pbpe_source": {
              "enum": ["co2", "soil"]
            },
            "coupon_structure": {
              "type": "string",
              "description": "Coupon structure linked to PBPE (e.g., 'pbpe_indexed').",
              "examples": ["pbpe_indexed"]
            },
            "min_co2_reduction_tons": {
              "type": "number",
              "description": "Minimum CO₂ reduction (tons) required for full payout."
            }
          }
        }
      ]
    },

    "PBPERegenerativeAgricultureBond": {
      "allOf": [
        { "$ref": "#/definitions/PBPEProductBase" },
        {
          "type": "object",

              "enum": ["soil", "yield", "water"]

          }
        }
      ]
    },

    "PBPEHealthImpactBond": {
      "allOf": [
        { "$ref": "#/definitions/PBPEProductBase" },
        {

              "const": "bond"

            "underlying_intent": {
              "const": "health"
            },
            "pbpe_source": {
              "enum": ["mbt55", "medical_cost"]
            },
            "min_medical_cost_reduction": {
              "type": "number",
              "description": "Minimum medical cost reduction required for full payout."
            },
            "target_population_size": {
              "type": "integer",
              "description": "Number of participants covered by the health program."
            }
          }
        }
      ]
    },

    "PBPECredit": {
      "allOf": [
        { "$ref": "#/definitions/PBPEProductBase" },
        {
          "type": "object",

            "product_type": {
              "const": "credit"

            "credit_type": {
              "type": "string",

                "carbon_credit",

              "type": "number",

    "PBPEInsurance": {

              "type": "string",

            "max_payout": {
              "type": "number",
              "description": "Maximum payout amount in the specified currency."
            }
          }
        }
      ]
    }
  }
}
```
